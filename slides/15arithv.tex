\begin{frame}{Representação de Números em Verilog}
    Seguem o formato <\#bits>'<base><valor>
    \begin{itemize}
        \item {\tt 12'b100010101001}
        \item {\tt 12'o4251}
        \item {\tt 12'h8A9}
        \item {\tt 12'd2217}
    \end{itemize}
\end{frame}
